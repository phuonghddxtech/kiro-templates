# Multi-Language (i18n) — React Native

> Internationalization and localization standards for React Native apps.

## Approved Stack

| Layer | Primary Choice | Alternative |
|-------|---------------|-------------|
| **i18n Library** | `expo-localization` + `i18next` + `react-i18next` | `expo-localization` + custom |
| **Translation Files** | JSON (per locale) | — |
| **Date/Time** | `date-fns` with locale | `dayjs` |
| **Number/Currency** | `Intl.NumberFormat` | — |
| **RTL Support** | `I18nManager` (React Native built-in) | — |

---

## Setup

```bash
npx expo install expo-localization i18next react-i18next
```

```ts
// src/i18n/index.ts
import i18n from 'i18next';
import { initReactI18next } from 'react-i18next';
import { getLocales } from 'expo-localization';

import vi from './locales/vi.json';
import en from './locales/en.json';

const deviceLocale = getLocales()[0]?.languageCode ?? 'vi';

i18n.use(initReactI18next).init({
  resources: {
    vi: { translation: vi },
    en: { translation: en },
  },
  lng: deviceLocale,
  fallbackLng: 'vi',
  interpolation: {
    escapeValue: false, // React Native already safe from XSS
  },
});

export default i18n;
```

### Import in App Entry
```tsx
// app/_layout.tsx
import '@/i18n'; // Initialize i18n before any component renders
```

## Usage in Components

```tsx
import { useTranslation } from 'react-i18next';

export function OrdersScreen() {
  const { t } = useTranslation();

  if (isLoading) return <LoadingState message={t('common.loading')} />;
  if (error) return <ErrorState message={t('errors.loadFailed', { resource: t('orders.title') })} onRetry={refetch} />;

  return (
    <View>
      <FlatList
        data={data}
        ListEmptyComponent={
          <EmptyState
            title={t('orders.empty')}
            message={t('orders.emptyMessage')}
          />
        }
      />
    </View>
  );
}
```

## Type Safety

```ts
// src/i18n/types.ts
import vi from './locales/vi.json';

// Auto-generate types from default locale
type TranslationKeys = typeof vi;

// Flatten nested keys for autocomplete: 'common.loading', 'orders.title', etc.
type FlattenKeys<T, Prefix extends string = ''> = {
  [K in keyof T]: T[K] extends object
    ? FlattenKeys<T[K], `${Prefix}${K & string}.`>
    : `${Prefix}${K & string}`;
}[keyof T];

export type TKey = FlattenKeys<TranslationKeys>;
```

## Date & Number Formatting

```ts
// utils/format.ts
import { format as fnsFormat } from 'date-fns';
import { vi, enUS } from 'date-fns/locale';
import i18n from '@/i18n';

const LOCALE_MAP = { vi, en: enUS };

export function formatDate(date: Date | string, pattern = 'dd/MM/yyyy'): string {
  const locale = LOCALE_MAP[i18n.language as keyof typeof LOCALE_MAP] ?? vi;
  return fnsFormat(new Date(date), pattern, { locale });
}

export function formatCurrency(amount: number): string {
  const locale = i18n.language === 'vi' ? 'vi-VN' : 'en-US';
  const currency = i18n.language === 'vi' ? 'VND' : 'USD';
  return new Intl.NumberFormat(locale, { style: 'currency', currency }).format(amount);
}

export function formatNumber(num: number): string {
  const locale = i18n.language === 'vi' ? 'vi-VN' : 'en-US';
  return new Intl.NumberFormat(locale).format(num);
}
```

## Language Switching

```ts
// stores/app.store.ts
import { create } from 'zustand';
import { MMKV } from 'react-native-mmkv';
import i18n from '@/i18n';

const storage = new MMKV();

interface AppStore {
  language: string;
  setLanguage: (lang: string) => void;
}

export const useAppStore = create<AppStore>((set) => ({
  language: storage.getString('app_language') ?? i18n.language,

  setLanguage: (lang) => {
    i18n.changeLanguage(lang);
    storage.set('app_language', lang);
    set({ language: lang });
  },
}));
```

```tsx
// Settings screen — language picker
const { language, setLanguage } = useAppStore();

<LanguagePicker
  value={language}
  options={[
    { label: 'Tiếng Việt', value: 'vi' },
    { label: 'English', value: 'en' },
  ]}
  onChange={setLanguage}
/>
```

## RTL Support

```tsx
// For Arabic, Hebrew, etc.
import { I18nManager } from 'react-native';
import * as Updates from 'expo-updates';

function setRTL(isRTL: boolean) {
  if (I18nManager.isRTL !== isRTL) {
    I18nManager.allowRTL(isRTL);
    I18nManager.forceRTL(isRTL);
    Updates.reloadAsync(); // Restart app to apply RTL
  }
}
```

## Rules

### ✅ MUST
- **NEVER hardcode strings** in components — always use `t('key')`
- **Vietnamese (`vi`) is the default locale** — all keys must exist in `vi.json`
- **Every translation file must have ALL keys** — no missing translations
- Use **interpolation** for dynamic values: `t('errors.loadFailed', { resource: 'đơn hàng' })`
- Use **nested namespaces** for organization: `common.*`, `auth.*`, `orders.*`
- **Numbers, dates, currency** — always format with locale-aware functions
- **Accessibility labels** must also be translated

### ❌ NEVER
- Never concatenate translated strings: `t('hello') + ' ' + name` → use `t('hello', { name })`
- Never use translation keys as display values
- Never hardcode date/number formats — use `formatDate()`, `formatCurrency()`

### Accessibility + i18n
```tsx
// ✅ Translate accessibility labels too
<Pressable
  accessibilityRole="button"
  accessibilityLabel={t('orders.create')}
  onPress={handleCreate}
>
  <Text>{t('orders.create')}</Text>
</Pressable>
```
